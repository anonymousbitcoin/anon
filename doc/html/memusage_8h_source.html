53 static inline size_t MallocUsage(size_t alloc)
54 {
55  // Measured on libc6 2.19 on Linux.
56  if (alloc == 0) {
57  return 0;
58  } else if (sizeof(void*) == 8) {
59  return ((alloc + 31) >> 4) << 4;
60  } else if (sizeof(void*) == 4) {
61  return ((alloc + 15) >> 3) << 3;
62  } else {
63  assert(0);
64  }
65 }
